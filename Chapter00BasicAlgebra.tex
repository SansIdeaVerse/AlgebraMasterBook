\section {Problem Set 02}
        \begin{enumerate}
            \item If $\dfrac{a}{1-a}+\dfrac{b}{1-b}+\dfrac{c}{1-c}=1$, then evaluate $\dfrac{1}{1-a}+\dfrac{1}{1-b}+\dfrac{1}{1-c}$.
            \item If $\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}=1$, then evaluate $\dfrac{a^2}{b+c}+\dfrac{b^2}{c+a}+\dfrac{c^2}{a+b}$.
            \item If $ab+bc+ca=0$, then evaluate $\dfrac{1}{a^2-bc}+\dfrac{1}{b^2-ca}+\dfrac{1}{c^2-ab}$.
            \item If $x+y+z=xyz$, then evaluate $\dfrac{1+x^2}{(x+y)(x+z)}+\dfrac{1+y^2}{(y+z)(y+x)}+\dfrac{1+z^2}{(z+x)(z+y)}$.
            \item If $pqr=1$, then evaluate $\dfrac{1}{1+p+q^{-1}}+\dfrac{1}{1+q+r^{-1}}+\dfrac{1}{1+r+p^{-1}}$. 
            \item Simplify: $\left(1+\dfrac{1}{x}\right)\left(1+\dfrac{1}{x-1}\right)\left(1+\dfrac{1}{x-2}\right)\cdots\left(1+\dfrac{1}{x-n}\right)$.
            \item If $a^2b+b^2c+c^2a=0$, then evaluate $\left( \dfrac{c}{a}+\dfrac{a}{b}\right) \left( \dfrac{a}{b}+\dfrac{b}{c}\right) \left( \dfrac{b}{c}+\dfrac{c}{a}\right)$.
            \item If $\dfrac{a^2-bc}{a^2+bc}+\dfrac{b^2-ca}{b^2+ca}+\dfrac{c^2-ab}{c^2+ab}=1$, then evaluate $\dfrac{a^2}{a^2+bc}+\dfrac{b^2}{b^2+ca}+\dfrac{c^2}{c^2+ab}$.
            \item If $\dfrac{a^2-bc}{a^2+bc}+\dfrac{b^2-ca}{b^2+ca}+\dfrac{c^2-ab}{c^2+ab}=1$, then evaluate $\dfrac{bc}{a^2+bc}+\dfrac{ca}{b^2+ca}+\dfrac{ab}{c^2+ab}$.









            % Express in terms of other variables
            \item $v=u+ft$. Express $f$ in terms of $v$, $u$, and $t$.
            \item $I=\dfrac{nE}{R+nr}$. Express $n$ in terms of $I$, $E$, $R$, and $r$.
            \item If $\dfrac{1}{f}=\dfrac{1}{u}+\dfrac{1}{v}$, then express $f$ in terms of $u$ and $v$.



            % Proof Section
            \item If $x=bz+cy$, $y=az+cx$, $z=ay+bx$, then prove that $\dfrac{x^2}{1-a^2}=\dfrac{y^2}{1-b^2}=\dfrac{z^2}{1-c^2}$.
            \item If $xy+yz+zx=1$, then prove that $\dfrac{2x}{1-x^2}+\dfrac{2y}{1-y^2}+\dfrac{2z}{1-z^2}=\dfrac{2x}{1-x^2}\cdot\dfrac{2y}{1-y^2}\cdot\dfrac{2z}{1-z^2}$.
            \item If $(b+c-a)x=(c+a-b)y=(a+b-c)z=2$, then prove that $$\left(\dfrac{1}{x}+\dfrac{1}{y}\right)\left(\dfrac{1}{y}+\dfrac{1}{z}\right)\left(\dfrac{1}{z}+\dfrac{1}{x}\right)=abc$$.
            \item If $\dfrac{a+b}{b+c}=\dfrac{c+d}{d+a}$, then prove that either $a=c$ or $a+b+c+d=0$.
            \item If a,b,c are continued proportion, then prove that $a^2b^2c^2\left(\dfrac{1}{a^3}+\dfrac{1}{b^3}+\dfrac{1}{c^3}\right)=a^3+b^3+c^3$.
            \item If $x+\dfrac{1}{x}=2a$, $x-\dfrac{1}{x}=2b$, $y+\dfrac{1}{y}=2c$, $y-\dfrac{1}{y}=2d$, then prove that $xy+\dfrac{1}{xy}=2(ca+bd)$.
            \item If mean proportional of x and z is y, then prove that mean proportional of $(x^2+y^2)$ and $(y^2+z^2)$ is $\left|(xy+yz)\right|$.

        \end{enumerate}
