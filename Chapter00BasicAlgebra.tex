        \begin{enumerate}
            \item If $\dfrac{a}{1-a}+\dfrac{b}{1-b}+\dfrac{c}{1-c}=1$, then evaluate $\dfrac{1}{1-a}+\dfrac{1}{1-b}+\dfrac{1}{1-c}$.

            \item If $ab+bc+ca=0$, then evaluate $\dfrac{a^2}{a^2-bc}+\dfrac{b^2}{b^2-ca}+\dfrac{c^2}{c^2-ab}$.
            \item If $ab+bc+ca=0$, then evaluate $\dfrac{bc}{a^2-bc}+\dfrac{ca}{b^2-ca}+\dfrac{ab}{c^2-ab}$.
            \item If $x+y+z=xyz$, then evaluate $\dfrac{1+x^2}{(x+y)(x+z)}+\dfrac{1+y^2}{(y+z)(y+x)}+\dfrac{1+z^2}{(z+x)(z+y)}$.
            \item If $pqr=1$, then evaluate $\dfrac{1}{1+p+q^{-1}}+\dfrac{1}{1+q+r^{-1}}+\dfrac{1}{1+r+p^{-1}}$. 
            \item Simplify: $\left(1+\dfrac{1}{x}\right)\left(1+\dfrac{1}{x-1}\right)\left(1+\dfrac{1}{x-2}\right)\cdots\left(1+\dfrac{1}{x-n}\right)$.
            \item If $a^2b+b^2c+c^2a=0$, then evaluate $\left( \dfrac{c}{a}+\dfrac{a}{b}\right) \left( \dfrac{a}{b}+\dfrac{b}{c}\right) \left( \dfrac{b}{c}+\dfrac{c}{a}\right)$.
            \item If $\dfrac{a^2-bc}{a^2+bc}+\dfrac{b^2-ca}{b^2+ca}+\dfrac{c^2-ab}{c^2+ab}=1$, then evaluate $\dfrac{a^2}{a^2+bc}+\dfrac{b^2}{b^2+ca}+\dfrac{c^2}{c^2+ab}$.
            \item If $\dfrac{a^2-bc}{a^2+bc}+\dfrac{b^2-ca}{b^2+ca}+\dfrac{c^2-ab}{c^2+ab}=1$, then evaluate $\dfrac{bc}{a^2+bc}+\dfrac{ca}{b^2+ca}+\dfrac{ab}{c^2+ab}$.
            \item If $x=by+cz+du$, $y=ax+cz+du$, $z=ax+by+du$, $u=ax+by+cz$, Evaluate the following:
                \begin{enumerate}[label=(\alph*)]

                    \item $\dfrac{1-a}{1+a}+\dfrac{1-b}{1+b}+\dfrac{1-c}{1+c}+\dfrac{1-d}{1+d}$
                \end{enumerate}

            \item If $x+y+z=t$ and $xy+yz+zx=2t-3$,then evaluate the followings:
                \begin{enumerate}[label=(\alph*)]
                    \item $\dfrac{1}{1-x}+\dfrac{1}{1-y}+\dfrac{1}{1-z}$
                    \item $\dfrac{1+x}{1-x}+\dfrac{1+y}{1-y}+\dfrac{1+z}{1-z}$
                \end{enumerate}
            \item If $x+y+z=6$ and $xy+yz+zx=9$,then evaluate the followings:
                \begin{enumerate}[label=(\alph*)]
                    \item $\dfrac{1}{1-x}+\dfrac{1}{1-y}+\dfrac{1}{1-z}$
                    \item $\dfrac{1+x}{1-x}+\dfrac{1+y}{1-y}+\dfrac{1+z}{1-z}$
                    \item $\dfrac{x^2}{1-x}+\dfrac{y^2}{1-y}+\dfrac{z^2}{1-z}$
                    \item $\dfrac{x^3}{1-x}+\dfrac{y^3}{1-y}+\dfrac{z^3}{1-z}$
